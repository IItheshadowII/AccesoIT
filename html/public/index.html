Python Reporte automatizado de ventas 
# Generar reporte diario de ventas
import pandas as pd
from datetime import datetime

def generate_sales_report():
    # Obtener datos de ventas
    sales_data = get_daily_sales()
    
    # Procesar datos y generar reporte
    df = pd.DataFrame(sales_data)
    report = df.groupby('categoria').agg({
        'cantidad': 'sum',
        'monto': 'sum'
    })
    
    # Guardar y enviar por email
    filename = f"ventas_{datetime.now().strftime('%Y%m%d')}.xlsx"
    report.to_excel(filename)
    send_email_report(filename)
